def process_subdir(subdir):
    """
    Process each individual subdirectory to separate JSON and media files.

    :param subdir: Path to the subdirectory to process.
    :return: Two lists of JSON and media files.
    """
    json_files = []
    media_files = []

    for filename in os.listdir(subdir):
        if filename.endswith('.json'):
            json_files.append({'subdir': subdir, 'file': filename.replace('\uf022', ':')})
        elif filename.endswith(('.jpeg', '.mp4')):
            media_files.append({'subdir': subdir, 'file': filename.replace('\uf022', ':')})

    return json_files, media_files

# Function to extract the item number from the filename
def extract_item_number(filename):
    # Attempt to extract the item number from the filename
    match = re.search(r'item(\d+)', filename)
    # Return the integer value of the item number if found, else None
    return (match is not None, int(match.group(1)) if match else 0)

def process_file_list(data_list):
    """
    Processes a list of dictionaries containing 'subdir' and 'file' keys.
    Replaces Unicode character '\uf022' with ':' in the 'file' values.
    Groups files by their 'subdir' into a dictionary and sorts files by item number.

    :param data_list: List of dictionaries with 'subdir' and 'file' keys.
    :return: Dictionary with 'subdir' as keys and lists of sorted 'file' names as values.
    """
    files = {}


    for item in data_list:
      if not item['subdir'] in files:
        files[item['subdir']] = []

      files[item['subdir']].append(item['file'])

    for subdir in files:
      files[subdir].sort(key=extract_item_number)

    return files
